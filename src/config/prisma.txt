datasource db {
  provider = "mysql"
  url      = env("DATABASE_URL")
}

model User {
  id        Int      @id @default(autoincrement())
  username      String?
  email     String   @unique
  password  String
  products  Product[]
  role      Role @default(USER)
  createdAt DateTime @default(now())
  updatedAt DateTime? @updatedAt

}

enum Role {
  ADMIN
  EMPLOYER
  USER
}

model Product {
  id          Int      @id @default(autoincrement())
  title       String
  description String?
  price       Float
  inStock     Boolean  @default(true)
  category    Category @default(None)
  stockId     Int?     // optional relation (nullable if not always linked)
  stock       Stock?   @relation(fields: [stockId], references: [id])
  userId      Int
  user        User     @relation(fields: [userId], references: [id])
  createdAt   DateTime @default(now())
  updateAt    DateTime @updatedAt
}

model Stock {
  id       Int       @id @default(autoincrement())
  count Int @default(0)
  category Category  @default(None)
  product Product[]
}

enum Category {
  MacBookAir
  MacBookPro
  iPhone
  iPad
  AirPods
  AppleWatchMini
  AppleWatchPro
  AppleWatchUltra
  Accessories
  None
}
